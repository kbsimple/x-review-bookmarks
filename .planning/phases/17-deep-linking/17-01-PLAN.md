function renderCardFooter(post) {
  const xUrl = `https://x.com/i/web/status/${esc(post.x_post_id)}`;
  const shareId = `share-${esc(post.x_post_id)}`;
  return `<div class="card-footer">

    <a href="${xUrl}" target="_blank" rel="noopener noreferrer" class="view-on-x">View on X</a>
  </div>`;

function copyDeepLink(postId, btnId) {
  const url = window.location.origin + window.location.pathname + '#post-' + postId;
  navigator.clipboard.writeText(url).then(function() {
    const btn = document.getElementById(btnId);
    if (btn) {
      btn.textContent = 'Copied!';
      setTimeout(function() { btn.innerHTML = '&#128100;'; }, 1500);
    }
  }).catch(function() {
    // clipboard denied in non-HTTPS context — silent fail
  });
}

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| URL hash → JS | `window.location.hash` value enters JS; postId extracted and used in DOM writes |
| Post data → HTML | `post.x_post_id` inserted into onclick attribute and button id via `esc()` |
| goHome target | `window.location.origin + window.location.pathname` — origin comes from browser-controlled property |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-17-01-01 | Tampering | `showDeepLinkError()` innerHTML | mitigate | Apply `esc(postId)` before inserting hash-extracted ID into innerHTML — implemented in showDeepLinkError action |
| T-17-01-02 | Tampering | `renderCardFooter()` onclick attribute | mitigate | Apply `esc(post.x_post_id)` on both the button id and onclick parameter — specified in Task 2 Edit 2 |
| T-17-01-03 | Tampering | `copyDeepLink()` URL construction | accept | `window.location.origin + window.location.pathname` uses browser-controlled properties; postId is appended after `#post-` which is client-side only — no server injection possible; X post IDs are numeric strings in practice |
| T-17-01-04 | Tampering | open redirect via `goHome()` | accept | `window.location.href = window.location.origin + window.location.pathname` hard-codes the current origin; no external input used in URL construction |
| T-17-01-05 | Information Disclosure | clipboard content | accept | Clipboard is user-initiated (explicit click); content is a public Netlify URL — no PII |
